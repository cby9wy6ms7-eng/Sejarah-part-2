        let currentCategory = ""; 
        let activeQuestions = []; 
        let currentQuestionIdx = 0;
        let userAnswers = []; 
        let quizStartTime = null;
        let timerInterval = null;
        let studentData = { name: "", class: "" };

        // STEP 1: MULAI KUIS BERDASARKAN PILIHAN
        function startSelectedQuiz() {
            const nameInput = document.getElementById("student-name").value.trim();
            const classSelect = document.getElementById("student-class").value;
            const categorySelect = document.querySelector('input[name="quiz-category"]:checked').value;

            if (!nameInput) return;

            studentData.name = nameInput;
            studentData.class = classSelect;
            currentCategory = categorySelect;

            // Load questions dari database
            activeQuestions = dataset[categorySelect];
            userAnswers = new Array(activeQuestions.length).fill(null);

            // Set UI header & display profile
            document.getElementById("profile-display-name").innerText = studentData.name;
            document.getElementById("profile-display-class").innerText = "Kelas: " + studentData.class;
            document.getElementById("total-questions-count").innerText = activeQuestions.length + " Total Soal";

            // Atur nama modul
            let modLabel = "";
            if (categorySelect === "pengayaan_3") modLabel = "Pengayaan Bab III";
            else if (categorySelect === "uh_3") modLabel = "UH Bab III";
            else if (categorySelect === "pengayaan_4") modLabel = "Pengayaan Bab IV";
            else if (categorySelect === "uh_4") modLabel = "UH Bab IV";
            document.getElementById("quiz-category-display").innerText = modLabel;

            // Sembunyikan layar awal, tampilkan layar kuis aktif
            document.getElementById("screen-start").classList.add("hidden");
            document.getElementById("screen-quiz").classList.remove("hidden");
            document.getElementById("badge-timer").classList.remove("hidden");

            // Mulai Timer
            quizStartTime = new Date();
            startTimer();

            // Buat Grid Navigasi sesuai jumlah soal modul terpilih
            renderJumpGrid();

            // Tampilkan soal pertama
            loadQuestion(0);
        }

        // TIMER HANDLER
        function startTimer() {
            if (timerInterval) clearInterval(timerInterval);
            timerInterval = setInterval(() => {
                const now = new Date();
                const diff = now - quizStartTime;
                const totalSeconds = Math.floor(diff / 1000);
                const minutes = Math.floor(totalSeconds / 60);
                const seconds = totalSeconds % 60;
                
                const timeString = 
                    (minutes < 10 ? "0" + minutes : minutes) + ":" + 
                    (seconds < 10 ? "0" + seconds : seconds);
                
                document.getElementById("quiz-timer").innerText = timeString;
            }, 1000);
        }

        // RENDER GRID NOMOR NAVIGASI CEPAT
        function renderJumpGrid() {
            const gridContainer = document.getElementById("jump-grid-container");
            gridContainer.innerHTML = "";

            for (let i = 0; i < activeQuestions.length; i++) {
                const btn = document.createElement("button");
                btn.id = `grid-btn-${i}`;
                btn.className = "h-8 w-full rounded-lg font-bold text-xs flex items-center justify-center transition border border-slate-200 bg-slate-50 text-slate-500 hover:bg-slate-100 focus:outline-none";
                btn.innerText = i + 1;
                btn.onclick = () => jumpToQuestion(i);
                gridContainer.appendChild(btn);
            }
        }

        // RE-HIGHLIGHT NAVIGASI GRID
        function updateJumpGridStatus() {
            for (let i = 0; i < activeQuestions.length; i++) {
                const btn = document.getElementById(`grid-btn-${i}`);
                if (!btn) continue;

                const hasAnswered = userAnswers[i] !== null;
                const isCurrent = (i === currentQuestionIdx);

                if (isCurrent) {
                    btn.className = "h-8 w-full rounded-lg font-extrabold text-xs flex items-center justify-center transition focus:outline-none bg-madrasah-600 text-white shadow-md shadow-madrasah-100";
                } else if (hasAnswered) {
                    btn.className = "h-8 w-full rounded-lg font-bold text-xs flex items-center justify-center transition focus:outline-none bg-blue-100 text-blue-700 hover:bg-blue-200";
                } else {
                    btn.className = "h-8 w-full rounded-lg font-bold text-xs flex items-center justify-center transition focus:outline-none border border-slate-200 bg-slate-50 text-slate-500 hover:bg-slate-100";
                }
            }
        }

        // TAMPILKAN SOAL KE CARD
        function loadQuestion(idx) {
            currentQuestionIdx = idx;
            const q = activeQuestions[idx];

            document.getElementById("quiz-question-number").innerText = "Soal " + (idx + 1);
            document.getElementById("quiz-question-id-tag").innerText = `Butir: ${q.id}`;
            document.getElementById("quiz-question-text").innerText = q.question;

            const container = document.getElementById("quiz-options-container");
            container.innerHTML = "";

            const hasAnswered = userAnswers[idx] !== null;

            for (let opt in q.options) {
                const optText = q.options[opt];
                const btn = document.createElement("button");
                btn.className = "w-full text-left p-3.5 rounded-xl border border-slate-200 hover:bg-slate-50 transition-all flex items-start space-x-3 text-sm focus:outline-none";
                btn.id = `opt-${opt}`;

                const letterBadge = document.createElement("span");
                letterBadge.className = "flex-shrink-0 w-6 h-6 rounded-full border border-slate-300 bg-white flex items-center justify-center font-bold text-xs text-slate-500 shadow-sm transition";
                letterBadge.innerText = opt;

                const textSpan = document.createElement("span");
                textSpan.className = "text-slate-700 leading-relaxed font-medium";
                textSpan.innerText = optText;

                btn.appendChild(letterBadge);
                btn.appendChild(textSpan);

                // Jika sudah dijawab sebelumnya
                if (hasAnswered) {
                    btn.disabled = true;
                    const chosen = userAnswers[idx];
                    const correct = q.correct;

                    if (opt === correct) {
                        btn.className = "w-full text-left p-3.5 rounded-xl border-2 border-green-600 bg-green-50/50 flex items-start space-x-3 text-sm font-semibold";
                        letterBadge.className = "flex-shrink-0 w-6 h-6 rounded-full bg-green-600 border border-transparent text-white flex items-center justify-center font-bold text-xs shadow-sm";
                    } else if (opt === chosen) {
                        btn.className = "w-full text-left p-3.5 rounded-xl border-2 border-red-500 bg-red-50/50 flex items-start space-x-3 text-sm";
                        letterBadge.className = "flex-shrink-0 w-6 h-6 rounded-full bg-red-500 border border-transparent text-white flex items-center justify-center font-bold text-xs shadow-sm";
                    } else {
                        btn.className = "w-full text-left p-3.5 rounded-xl border border-slate-200 opacity-60 flex items-start space-x-3 text-sm pointer-events-none";
                    }
                } else {
                    btn.onclick = () => selectOption(opt);
                }

                container.appendChild(btn);
            }

            // Atur Pembahasan Umpan Balik Instan
            const feedbackBox = document.getElementById("quiz-feedback-box");
            if (hasAnswered) {
                const chosen = userAnswers[idx];
                const correct = q.correct;
                const isCorrect = (chosen === correct);

                feedbackBox.classList.remove("hidden");
                const headingEl = document.getElementById("quiz-feedback-heading");
                const iconEl = document.getElementById("quiz-feedback-icon");
                const rationaleEl = document.getElementById("quiz-feedback-rationale");

                rationaleEl.innerText = q.rationale;

                if (isCorrect) {
                    feedbackBox.className = "rounded-xl p-4 transition-all duration-300 bg-green-50 border border-green-200 text-green-900";
                    headingEl.innerText = "Kunci Jawaban Benar!";
                    iconEl.className = "p-1.5 bg-green-100 text-green-700 rounded-lg mt-0.5";
                    iconEl.innerHTML = `<svg class="w-4 h-4" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2.5" d="M9 12l2 2 4-4m6 2a9 9 0 11-18 0 9 9 0 0118 0z"></path></svg>`;
                } else {
                    feedbackBox.className = "rounded-xl p-4 transition-all duration-300 bg-red-50 border border-red-200 text-red-900";
                    headingEl.innerText = `Jawaban Kurang Tepat (Pilihan Anda: ${chosen}, Kunci: ${correct})`;
                    iconEl.className = "p-1.5 bg-red-100 text-red-700 rounded-lg mt-0.5";
                    iconEl.innerHTML = `<svg class="w-4 h-4" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2.5" d="M10 14l2-2m0 0l2-2m-2 2l-2-2m2 2l2 2m7-2a9 9 0 11-18 0 9 9 0 0118 0z"></path></svg>`;
                }
            } else {
                feedbackBox.classList.add("hidden");
            }

            // Atur State Tombol Prev/Next
            document.getElementById("btn-prev").disabled = (idx === 0);
            const btnNextText = document.getElementById("btn-next-text");
            if (idx === activeQuestions.length - 1) {
                btnNextText.innerText = "Kumpul";
            } else {
                btnNextText.innerText = "Selanjutnya";
            }

            // Refresh progress display
            updateJumpGridStatus();
            updateProgressBar();
        }

        // SAAT OPSI JAWABAN DIPILIH
        function selectOption(selectedOpt) {
            userAnswers[currentQuestionIdx] = selectedOpt;
            loadQuestion(currentQuestionIdx); // Refresh untuk langsung memunculkan tanda warna & rationale
        }

        // TOMBOL NAVIGASI BERIKUTNYA / SEBELUMNYA
        function prevQuestion() {
            if (currentQuestionIdx > 0) {
                loadQuestion(currentQuestionIdx - 1);
            }
        }

        function nextQuestion() {
            if (currentQuestionIdx < activeQuestions.length - 1) {
                loadQuestion(currentQuestionIdx + 1);
            } else {
                confirmFinishQuiz();
            }
        }

        function jumpToQuestion(idx) {
            loadQuestion(idx);
        }

        // UPDATE PROGRESS BAR
        function updateProgressBar() {
            const answeredCount = userAnswers.filter(a => a !== null).length;
            const percentage = Math.round((answeredCount / activeQuestions.length) * 100);

            document.getElementById("progress-percentage").innerText = percentage + "%";
            document.getElementById("progress-bar-fill").style.width = percentage + "%";
            document.getElementById("answered-count").innerText = answeredCount + " Terjawab";
        }

        // CONFIRM MODAL SEBELUM SELESAI
        function confirmFinishQuiz() {
            const answeredCount = userAnswers.filter(a => a !== null).length;
            const unansweredCount = activeQuestions.length - answeredCount;

            document.getElementById("modal-answered").innerText = answeredCount + " Soal";
            document.getElementById("modal-unanswered").innerText = unansweredCount + " Soal";

            document.getElementById("modal-confirm").classList.remove("hidden");
        }

        function closeModal() {
            document.getElementById("modal-confirm").classList.add("hidden");
        }

        // SELESAI & HITUNG SKOR
        function finishQuiz() {
            closeModal();
            clearInterval(timerInterval);

            // Ganti layar ke layar hasil evaluasi
            document.getElementById("screen-quiz").classList.add("hidden");
            document.getElementById("screen-result").classList.remove("hidden");
            document.getElementById("badge-timer").classList.add("hidden");

            // Hitung statistik jawaban sahih
            let correctCount = 0;
            for (let i = 0; i < activeQuestions.length; i++) {
                if (userAnswers[i] === activeQuestions[i].correct) {
                    correctCount++;
                }
            }

            // Kalkulasi nilai berbobot persentase dari jumlah soal total modul
            const totalScore = Math.round((correctCount / activeQuestions.length) * 100);
            const incorrectCount = activeQuestions.length - correctCount;
            const elapsedDuration = document.getElementById("quiz-timer").innerText;

            // Masukkan data ke halaman hasil
            document.getElementById("result-display-name").innerText = `Selamat, ${studentData.name}!`;
            document.getElementById("result-display-meta").innerText = `Anda telah merampungkan pengerjaan modul kuis dengan kategori pilihan Anda.`;
            document.getElementById("stat-correct").innerText = correctCount;
            document.getElementById("stat-incorrect").innerText = incorrectCount;
            document.getElementById("stat-time").innerText = elapsedDuration;

            // Set lingkaran skor SVG
            document.getElementById("score-display-number").innerText = totalScore;
            const ringFill = document.getElementById("score-ring-fill");
            const circumference = 2 * Math.PI * 54; // 339.29
            const strokeOffset = circumference - (totalScore / 100) * circumference;
            ringFill.style.strokeDashoffset = strokeOffset;

            // Label KKM Madrasah
            const kkmBadge = document.getElementById("kkm-status-badge");
            if (totalScore >= 85) {
                kkmBadge.className = "px-4 py-1.5 rounded-full text-[10px] font-extrabold tracking-wider uppercase bg-green-100 text-green-800 border border-green-200";
                kkmBadge.innerText = "PRESTASI SANGAT BAIK (LULUS)";
            } else if (totalScore >= 75) {
                kkmBadge.className = "px-4 py-1.5 rounded-full text-[10px] font-extrabold tracking-wider uppercase bg-blue-100 text-blue-800 border border-blue-200";
                kkmBadge.innerText = "LULUS KKM MADRASAH";
            } else {
                kkmBadge.className = "px-4 py-1.5 rounded-full text-[10px] font-extrabold tracking-wider uppercase bg-red-100 text-red-800 border border-red-200";
                kkmBadge.innerText = "TIDAK LULUS (BELUM KKM)";
            }

            // Tampilkan Kunci Jawaban Lengkap
            buildReviewPanel();
        }

        // GENERATE DAFTAR TINJAUAN JAWABAN (REVIEW PANEL)
        function buildReviewPanel() {
            const container = document.getElementById("review-list-container");
            container.innerHTML = "";

            activeQuestions.forEach((q, idx) => {
                const isCorrect = (userAnswers[idx] === q.correct);
                const chosen = userAnswers[idx] || "Tidak Dijawab";

                const card = document.createElement("div");
                card.className = `p-5 rounded-2xl border transition-all ${
                    isCorrect 
                        ? 'border-green-100 bg-green-50/20' 
                        : 'border-red-100 bg-red-50/20'
                }`;

                const header = document.createElement("div");
                header.className = "flex flex-wrap justify-between items-center gap-1.5 pb-2 border-b border-dashed border-slate-200 mb-3 text-xs";

                const meta = document.createElement("span");
                meta.className = "text-slate-400 font-semibold italic";
                meta.innerText = `Soal ${idx + 1} (Kode Butir: ${q.id})`;

                const badge = document.createElement("span");
                if (isCorrect) {
                    badge.className = "bg-green-100 text-green-800 font-bold px-2.5 py-0.5 rounded-md flex items-center space-x-1";
                    badge.innerHTML = `<svg class="w-3.5 h-3.5 mr-0.5" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2.5" d="M9 12l2 2 4-4"></path></svg> Benar`;
                } else {
                    badge.className = "bg-red-100 text-red-800 font-bold px-2.5 py-0.5 rounded-md flex items-center space-x-1";
                    badge.innerHTML = `<svg class="w-3.5 h-3.5 mr-0.5" fill="none" stroke="currentColor" viewBox="0 0 24 24"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2.5" d="M6 18L18 6M6 6l12 12"></path></svg> Salah / Kosong`;
                }

                header.appendChild(meta);
                header.appendChild(badge);

                const questionText = document.createElement("p");
                questionText.className = "font-bold text-slate-800 text-sm leading-relaxed mb-3";
                questionText.innerText = q.question;

                // Jawaban terpilih vs Kunci
                const detailsGrid = document.createElement("div");
                detailsGrid.className = "grid grid-cols-1 sm:grid-cols-2 gap-2 text-xs mb-3.5";

                const chosenBox = document.createElement("div");
                chosenBox.className = `p-2.5 rounded-xl border ${
                    isCorrect 
                        ? 'border-green-200 bg-green-100/50 text-green-900' 
                        : 'border-red-200 bg-red-100/50 text-red-900'
                }`;
                chosenBox.innerHTML = `<strong>Jawaban Anda:</strong> (${chosen}) ${q.options[chosen] || "Tidak dijawab"}`;

                const correctBox = document.createElement("div");
                correctBox.className = "p-2.5 rounded-xl border border-slate-200 bg-slate-100 text-slate-800";
                correctBox.innerHTML = `<strong>Kunci Sahih:</strong> (${q.correct}) ${q.options[q.correct]}`;

                detailsGrid.appendChild(chosenBox);
                detailsGrid.appendChild(correctBox);

                // Kotak Pembahasan (Rationale)
                const rationaleBox = document.createElement("div");
                rationaleBox.className = "p-3 bg-white/95 border border-slate-200 rounded-xl text-xs text-slate-600 leading-relaxed shadow-sm";
                rationaleBox.innerHTML = `<strong>Kunci Pembahasan:</strong> ${q.rationale}`;

                card.appendChild(header);
                card.appendChild(questionText);
                card.appendChild(detailsGrid);
                card.appendChild(rationaleBox);

                container.appendChild(card);
            });
        }

        // RESET KE HALAMAN DEPAN
        function backToMain() {
            document.getElementById("screen-result").classList.add("hidden");
            document.getElementById("screen-start").classList.remove("hidden");
            document.getElementById("student-name").value = "";
        }

        // SCROLL HELPER

            document.getElementById("review-section").scrollIntoView({ behavior: 'smooth' });
        }
